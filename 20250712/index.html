    int n;

    std::vector<int> a(n + 1), p(n + 1);

        std::cin >> a[i], p[i] = i;

    std::vector<int> f(n + 1), siz(n + 1, 1);

        return x == f[x] ? x : f[x] = find(f[x]);

        if (x != y)

        auto res(1ll);
        for (; y; (x *= x) %= mod, y >>= 1)

                (res *= x) %= mod;
        return res;
    };

            (res *= g[siz[i]]) %= mod;

        }
