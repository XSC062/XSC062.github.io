#include <bits/stdc++.h>
const int maxn = 3e5 + 5;
const int inf = 0x3f3f3f3f;
struct { int l, r, u, c, d; } t[maxn << 2];
#define lt (p << 1)
#define rt (lt | 1)
void pushup(int p) {
	t[p].u = std::min(t[lt].u, t[rt].u);
	t[p].c = 0;
	if (t[p].u == t[lt].u)
		t[p].c += t[lt].c;
	if (t[p].u == t[rt].u)
		t[p].c += t[rt].c;
	return;
}
void pushdown(int p) {
	if (t[p].d) {
		t[lt].d += t[p].d;
		t[rt].d += t[p].d;
		t[lt].u += t[p].d;
		t[rt].u += t[p].d;
		t[p].d = 0;
	}
	return;
}
void bld(int p, int l, int r) {
	t[p].l = l, t[p].r = r;
	if (l == r) {
		t[p].c = 1;
		return;
	}
	int mid = (l + r) >> 1;
	bld(lt, l, mid);
	bld(rt, mid + 1, r);
	pushup(p);
	return; 
}
void upd(int p, int l, int r, int v) {
	if (l <= t[p].l && t[p].r <= r) {
		t[p].u += v;
		t[p].d += v;
		return;
	}
	pushdown(p);
	int mid = (t[p].l + t[p].r) >> 1;
	if (l <= mid)
		upd(lt, l, r, v);
	if (r > mid)
		upd(rt, l, r, v);
	pushup(p);
	return;
}

}
