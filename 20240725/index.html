G. Pudding Monsters

https://www.becoder.com.cn/contest/5431/problem/7

省流：给定 \(a\)，问存在多少个 \([l, r]\)，满足 \(r-l=\max a_{l\sim r}-\min a_{l\sim r}\)。

考虑枚举右端点，在线段树上实时对满足 \(\max a-\min a-len=-1\) 的区间进行计数。区间维护 \(-1\) 的个数。

显然在题目限制下，\(\max a-\min a-len\) 的值最小即为 \(-1\)，故可以在线段树上合并信息的时候对最小值为 \(-1\) 的区间计数。当然因为这样看起来太奇怪了导致米娜桑大都采用「对拥有最小值的区间计数，如果最终最小值为 -1 再累加到答案」的方式实现（）

具体实现，显然一个新加入的右端点只能对所有后缀区间的最值产生影响，因此我们用单调栈来更新最值。顺手给所有区间做区间加来维护长度。

#include <bits/stdc++.h>
const int maxn = 3e5 + 5;
const int inf = 0x3f3f3f3f;
struct { int l, r, u, c, d; } t[maxn << 2];
#define lt (p << 1)
#define rt (lt | 1)
void pushup(int p) {
	t[p].u = std::min(t[lt].u, t[rt].u);
	t[p].c = 0;
	if (t[p].u == t[lt].u)
		t[p].c += t[lt].c;
	if (t[p].u == t[rt].u)
		t[p].c += t[rt].c;
	return;
}
void pushdown(int p) {
	if (t[p].d) {
		t[lt].d += t[p].d;
		t[rt].d += t[p].d;
		t[lt].u += t[p].d;
		t[rt].u += t[p].d;
		t[p].d = 0;
	}
	return;
}
void bld(int p, int l, int r) {
	t[p].l = l, t[p].r = r;
	if (l == r) {
		t[p].c = 1;
		return;
	}
	int mid = (l + r) >> 1;
	bld(lt, l, mid);
	bld(rt, mid + 1, r);
	pushup(p);
	return; 
}
void upd(int p, int l, int r, int v) {
	if (l <= t[p].l && t[p].r <= r) {
		t[p].u += v;
		t[p].d += v;
		return;
	}
	pushdown(p);
	int mid = (t[p].l + t[p].r) >> 1;
	if (l <= mid)
		upd(lt, l, r, v);
	if (r > mid)
		upd(rt, l, r, v);
	pushup(p);
	return;
}
int main() {
#ifdef ONLINE_JUDGE
	std::ios::sync_with_stdio(false);
	std::cin.tie(nullptr);
	std::cout.tie(nullptr);
#endif
	int n;
	std::cin >> n;
	std::vector<int> a(n);
	for (int i = 0, x; i < n; ++i) {
		std::cin >> x;
		std::cin >> a[x - 1];
	}
	bld(1, 0, n - 1);
	std::stack<int> ta, ti;
	ta.push(-1), ti.push(-1);
	long long res = 0;
	for (int i = 0; i < n; ++i) {
		while (~ta.top() && a[i] > a[ta.top()]) {
			int t = ta.top();
			ta.pop();
			upd(1, ta.top() + 1, t, -a[t]);
		}
		upd(1, ta.top() + 1, i, a[i]);
		ta.push(i);
		while (~ti.top() && a[i] < a[ti.top()]) {
			int t = ti.top();
			ti.pop();
			upd(1, ti.top() + 1, t, a[t]);
		}
		upd(1, ti.top() + 1, i, -a[i]);
		ti.push(i);
		upd(1, 0, i, -1);
		if (t[1].u == -1)
			res += t[1].c;
	}
	std::cout << res << '\n';
	return 0;
}

F - Good Subsegments

https://vjudge.net/contest/649698#problem/F

省流：给定 \(Q\) 个询问 \([l, r]\)，问满足 \(l\le x\le y\le r\) 且 \(y - x = \max a_{x\sim y}-\min a_{x\sim y}\) 的有序数对 \((x, y)\) 的数量。

Pudding Monsters 进化版。

考虑沿用 Pudding Monsters 中的套路，用线段树维护信息。

🌼 鲜花

我和智力为了怎么维护这玩意儿争论不休。为了结束争论，我富有洞见性地打开了题解。

然后两个人（一人一猴罢了）惊叹于本题充满扑面而来的原始气息的解决方式。

离线下来，按右端点从小到大处理问题。然后对于不在查询范围内左端点对答案造成的贡献，非常原始地解决掉——在线段树上顺便记录每个左端点造成的贡献，从答案中减去它们，或者更干脆地，只输出合法的贡献。

然后我们有一个很怪的实现方法就是打一个懒标记，表示「将当前区间贡献增加区间内 \(-1\) 的个数」，然后又知道每次修改 / 询问时都要下传懒标记，所以是对的。

注意到我们本来已经有一个区间加的标记了，直觉上来看两个标记不会相互影响，但是细想一下，我们的答案统计是基于最小值的，下传区间加标记会对子区间最小值带来影响，当子区间存在答案统计标记时就会有冲突。

其实进一步地细想，答案统计只关心 \(-1\) 的个数，对最小值的具体值反而没那么关心，我们关心的只是「在打标记时区间最小值是否为 \(-1\)」，而不是「当前时刻区间最小值是否为 \(-1\)」。所以我们只需要先下传区间加标记，在打标记和下传标记的时候只打给最小值为当前区间最小值的子区间即可。

为啥是最小值为当前区间最小值的子区间而非最小值为 \(-1\) 的子区间呢，因为我们这里下传的可能是之前某次的懒标记，兜兜转转几载春秋早已物是人非，故而只需保证和当前区间最小值一致。本题还有个性质是 \([r, r]\) 一定合法，故枚举每个右端点都有一个最小值 \(-1\)，所以可以放心地直接给树根打标记。

以及理论上（经实验实际上也是）来说一个区间是可能会被打多次答案统计标记的，所以该标记表示的应为「应统计的次数」而非「是否统计」，更新贡献的时候也记得要用次数乘以 \(-1\) 个数。

#include <bits/stdc++.h>
const int maxn = 1.2e5 + 5;
const int inf = 0x3f3f3f3f;
struct {
	long long t;
	int l, r, u, c, d, d1;
} t[maxn << 2];
int a[maxn];
long long res[maxn];
std::vector<std::pair<int, int> > u[maxn];
#define lt (p << 1)
#define rt (lt | 1)
void pushup(int p) {
	t[p].u = std::min(t[lt].u, t[rt].u);
	t[p].c = 0;
	if (t[p].u == t[lt].u)
		t[p].c += t[lt].c;
	if (t[p].u == t[rt].u)
		t[p].c += t[rt].c;
	return;
}
void pushdown(int p) {
	if (t[p].d) {
		t[lt].d += t[p].d;
		t[rt].d += t[p].d;
		t[lt].u += t[p].d;
		t[rt].u += t[p].d;
		t[p].d = 0;
	}
	if (t[p].d1) {
		if (t[lt].u == t[p].u) {
			t[lt].d1 += t[p].d1;
			t[lt].t += (long long)t[lt].c * t[p].d1;
		}
		if (t[rt].u == t[p].u) {
			t[rt].d1 += t[p].d1;
			t[rt].t += (long long)t[rt].c * t[p].d1;
		}
		t[p].d1 = 0;
	}
	return;
}
void bld(int p, int l, int r) {
	t[p].l = l, t[p].r = r;
	if (l == r) {
		t[p].c = 1;
		return;
	}
	int mid = (l + r) >> 1;
	bld(lt, l, mid);
	bld(rt, mid + 1, r);
	pushup(p);
	return; 
}
void upd(int p, int l, int r, int v) {
	if (l <= t[p].l && t[p].r <= r) {
		t[p].u += v;
		t[p].d += v;
		return;
	}
	pushdown(p);
	int mid = (t[p].l + t[p].r) >> 1;
	if (l <= mid)
		upd(lt, l, r, v);
	if (r > mid)
		upd(rt, l, r, v);
	pushup(p);
	return;
}
long long ask(int p, int l, int r) {
	if (l <= t[p].l && t[p].r <= r)
		return t[p].t;
	pushdown(p);
	long long res = 0;
	int mid = (t[p].l + t[p].r) >> 1;
	if (l <= mid)
		res = ask(lt, l, r);
	if (r > mid)
		res += ask(rt, l, r);
	return res;
}
int main() {
#ifdef ONLINE_JUDGE
	std::ios::sync_with_stdio(false);
	std::cin.tie(nullptr);
	std::cout.tie(nullptr);
#endif
	int n;
	std::cin >> n;
	for (int i = 1; i <= n; ++i)
		std::cin >> a[i];
	bld(1, 1, n);
	int q;
	std::cin >> q;
	for (int i = 1, l, r; i <= q; ++i) {
		std::cin >> l >> r;
		u[r].emplace_back(l, i);
	}
	std::stack<int> ta, ti;
	ta.push(0), ti.push(0);
	for (int i = 1; i <= n; ++i) {
		while (ta.top() && a[i] > a[ta.top()]) {
			int t = ta.top();
			ta.pop();
			upd(1, ta.top() + 1, t, -a[t]);
		}
		upd(1, ta.top() + 1, i, a[i]);
		ta.push(i);
		while (ti.top() && a[i] < a[ti.top()]) {
			int t = ti.top();
			ti.pop();
			upd(1, ti.top() + 1, t, a[t]);
		}
		upd(1, ti.top() + 1, i, -a[i]);
		ti.push(i);
		upd(1, 1, i, -1);
		++t[1].d1;
		t[1].t += t[1].c;
		for (auto j : u[i])
			res[j.second] += ask(1, j.first, i);
	}
	for (int i = 1; i <= q; ++i)
		std::cout << res[i] << '\n';
	return 0;
}

C - 网络

https://vjudge.net/contest/649698#problem/C

省流：给出一个带权树，现可任选两点连一条长度为 \(L\) 的边，最小化任意两点间最短路的最大值。

显然我们至少需要让当前的最大值变小，也就是说，让直径两个端点之间的距离变短。

然后我们就会有一个下意识的想法——答案选取的两个点一定都在直径上吗？

首先我们知道，若我们在 \((x, y)\) 间连边，那么可能影响到的只有和 \(x\to y\) 交集不为空的路径。若欲使直径两端点间距离变短，首先肯定不能连在直径上一点的
