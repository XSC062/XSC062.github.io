A. 色

做法一：最小生成树

需要注意到：

  • 最短路径一定是图上的一条边。反证：若为 \(x\to y\to z\)，若 \(col_x=col_y\)，则 \(y\to z\) 更优；否则，\(x\to y\) 更优，均矛盾。
  • 进一步地，一定是最小生成树上的某条边。归纳：在 Kruskal 时，若当前选中边组成的森林中，每一棵树中颜色均相同，加入一条连接着新颜色的新边，此时该边为该颜色到已有颜色的最短边。

然后注意到一堆人做到这里就不会了，提交了各式各样稀奇古怪的暴力，结果当然是被我和另一位匿名人士卡得底裤都不剩了 😂

一个简单的正解是平衡树，我不会故开摆。spdarkle 好像打的 set，不确定，等会儿看看。

做法二：分块

膜拜神 dj，膜拜神 nyh，不然我就真的只能去贺 fhq 板子了。

还是从最短路径一定是图上边出发，对边序列排序后分块。

分为两种情况：块中每条边都不可选（不存在边两端颜色不同）、块中有边可选（存在边两端颜色不同）。

询问时枚举所有块，如果该块中有边可选，则暴力枚举块中边然后 break。

修改和维护则是祖传哈希。给每条边赋一个随机系数并强行有向，起点为加终点为减记录每块权值，记录每个点在每个块中的系数和，最后总和为 \(0\) 则块不可选。

#include <bits/stdc++.h>
using ull = unsigned long long;
const int maxk = 555;
const int maxn = 2e5 + 5;
const int maxm = 3e5 + 5;
struct _ { int u, v, w; };
_ g[maxm];
int l[maxk], r[maxk], a[maxn];
ull hash[maxm], w[maxk], to[maxn], val[maxk][maxn];
int main() {
#ifdef ONLINE_JUDGE
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    std::cout.tie(nullptr);
#else
    freopen(".in", "r", stdin);
    freopen(".out", "w", stdout);
#endif
    int n, m, q;
    std::cin >> n >> m >> q >> q;
	std::mt19937_64 rnd(0xabcdef);
    for (int i = 1; i <= m; ++i) {
		std::cin >> g[i].u >> g[i].v >> g[i].w;
		hash[i] = rnd();
	}
	for (int i = 1; i <= n; ++i) {
		std::cin >> a[i];
		for (; !to[a[i]]; to[a[i]] = rnd());
	}
	std::sort(g + 1, g + m + 1, [](_ x, _ y) { return x.w < y.w; });
	int siz = sqrt(m), k = (m + siz - 1) / siz;
	for (int i = 1; i <= k; ++i) {
		l[i] = r[i - 1] + 1, r[i] = std::min(l[i] + siz - 1, m);
		for (int j = l[i]; j <= r[i]; ++j) {
			val[i][g[j].u] += hash[j];
			val[i][g[j].v] -= hash[j];
			w[i] += hash[j] * (to[a[g[j].u]] - to[a[g[j].v]]);
		}
	}
	for (int x, y; q--; ) {
		std::cin >> x >> y;
		for (; !to[y]; to[y] = rnd());
		for (int i = 1; i <= k; ++i)
			w[i] += val[i][x] * (to[y] - to[a[x]]);
		a[x] = y;
		for (int i = 1; i <= k; ++i) {
			// printf("w[%d] = %llu\n", i, w[i]);
			if (w[i]) {
				for (int j = l[i]; j <= r[i]; ++j)
					if (a[g[j].u] != a[g[j].v]) {
						std::cout << g[j].w << '\n';
						break;
					}
				break;
			}
		}
	}
    return 0;
}

E. Tom & Jerry

http://222.180.160.110:61234/contest/5618/problem/5

我们有一个很容易猜到的策略：如果 Tom 先站在一个割点上，每一步朝着 Jerry 的方向走到相邻的割点，那么就可以抓住 Jerry。

也就是说，我们对原图建立圆方树，若 Tom 在割点上，且以该割点为根，在 Jerry 所在的圆点对应的子树中，每个圆点到自己的所有方点儿子的所有圆点儿子在原图上都有一条直接连边，则 Tom 必胜。

那如果 Tom 不在割点上呢？由于 Jerry 先手，可以跑到任意一个，除非存在一个割点，满足以它为根，每个圆点到自己的所有方点儿子的所有圆点儿子在原图上都有一条直接连边，则 Tom 可以走到该必胜点。此时 Tom 必胜，与 Tom 和 Jerry 的初始位置无关。

总结起来，若存在必胜割点，则对于该组数据全部无脑 Yes 即可；否则，必须满足要求才可必胜。考虑 \(f_u ,g_u\) 分别表示 1 为根时 \(u\) 内 / 外子树合法性，然后随便换根即可。倍增一下找一找 Jerry 在 Tom 的哪个子树里就行。

笑话：虽然这里说的是随便换根即可，但实际上这个 trick 是看了题解才学会的。属于是狠狠地嘲笑了。

以及有个实现上的细节：求解 \(g\) 的时候势必需要知道对于圆方树上 \(u\) 的儿子 \(v_{1,2,\cdots}\) 中的每一个 \(v_i\)，有多少条 \(v_i\to v_j\) 在原图上存在。如果我们枚举 \(v_i,v_j\)，这个东西显然会是 \(n^2\) 的，那么我们反过来，枚举 \(v_i\) 在原图上的所有边 \(v_i\to p\)，判断 \(p\in v\) 是否成立即可。

被这个细节卡了 114514s，有被自己蠢到 😅

	std::vector<std::vector<int> > g1(n + 1);

		dfn[x] = low[x] = ++now;
		t.push(x);
		for (auto i : g1[x])
			if (!dfn[i]) {
				DFS(i, x);
				low[x] = std::min(low[x], low[i]);
				if (low[i] >= dfn[x]) {

					for (;;) {

						t.pop();
						g2[p].push_back(cnt), g2[cnt].push_back(p);
						if (p == i)
							break;
					}
					g2[x].push_back(cnt), g2[cnt].push_back(x);
				}
				if (x == 1)
					++rts;
				else if (low[i] >= dfn[x])
					tag[x] = 1;
			}
			else if (i != fa && dfn[i] < dfn[x])
				low[x] = std::min(low[x], dfn[i]);
		return;
	};
	DFS(1, -1);
	tag[1] = (rts >= 2);
	std::set<std::pair<int, int> > t;
	for (int i = 1; i <= n; ++i)
		if (tag[i])
			for (auto j : g1[i])
				if (tag[j] || (int)g2[j].size() == 1)
					t.emplace(i, j);
	dfn.assign(cnt + 1, 0);
	std::vector<std::vector<int> > fa(cnt + 1, std::vector<int> (21));
	std::vector<int> f(cnt + 1, 1), g(cnt + 1), rfn(cnt + 1), dep(cnt + 1);
	std::function<void(int)> DFS1 = [&](int x) {
		static int now = 0;
		dfn[x] = ++now;
		for (auto i : g2[x])
			if (i != fa[x][0]) {
				fa[i][0] = x;
				for (int j = 1; j <= 20; ++j)
					fa[i][j] = fa[fa[i][j - 1]][j - 1];
				dep[i] = dep[x] + 1;
				DFS1(i);
				f[x] *= f[i];
				if (x > n)
					f[x] *= t.count({ fa[x][0], i });
			}
		rfn[x] = now;
		return;
	};
	dep[1] = 1, DFS1(1);
	DFS1 = [&](int x) {
		int c0 = 0;
		std::set<int> t1;
		for (auto i : g2[x]) {
			if (i != fa[x][0])
				c0 += !f[i];
			if (x > n)
				t1.insert(i);
		}
		for (auto i : g2[x])
			if (i != fa[x][0]) {
				g[i] = g[x] * !(c0 - !f[i]);
				if (x > n) {

					for (auto j : g1[i])
						cnt += t1.count(j);

			std::cout << res[f[y]] << '\n';

B. party

http://222.180.160.110:61234/contest/5636/problem/2

集合点就是 LCA。不难想到是要树剖的，但线段树怎么维护信息呢？由于 \(m\) 很小，（很难）可以想到用 std::bitset<> 维护。

在知道每个点的信息之后如何完成判定呢？（很难）可以想到二分答案 \(x\) 然后把每个人拆成 \(x\) 个点，和其可以带的所有东西连边建出二分图，check 是否存在完美匹配。但这个复杂度太搞笑了，所以我们想到和「完美匹配」有关的 霍尔定理：

二分图 G 存在完美匹配 \(\iff\) 从左部任选 \(k(1\le k\le n)\) 个点，与它们直接相连的右部点至少有 \(k\) 个。

左部不同的连边实际上只有 \(c\) 种，我们 \(2^c\) 枚举子集 \(S\) 然后 std::bitset<> 暴力合并右部点 \(R\)，取 \(\min\left\{\left\lfloor\dfrac{|R|}{|S|}\right\rfloor\right\}\times c\)。怎么听起来感觉常数这么大啊。

#include <bits/stdc++.h>
const int maxn = 3e5 + 5;
const int inf = 0x3f3f3f3f;
struct {
	int l, r;
	std::bitset<1001> u;
} t[maxn << 2];
int tab[maxn], a[maxn];
#define lt (p << 1)
#define rt (lt | 1)
void pushup(int p) {
	t[p].u = t[lt].u | t[rt].u;
	return;
}
void bld(int p, int l, int r) {
	t[p].l = l, t[p].r = r;
	if (l == r) {
		t[p].u.set(a[tab[l]]);
		return;
	}
	int mid = (l + r) >> 1;
	bld(lt, l, mid), bld(rt, mid + 1, r);
	pushup(p);
	return;
}
std::bitset<1001> ask(int p, int l, int r) {
	if (l <= t[p].l && t[p].r <= r)
		return t[p].u;
	std::bitset<1001> res;
	int mid = (t[p].l + t[p].r) >> 1;
	if (l <= mid)
		res = ask(lt, l, r);
	if (r > mid)
		res |= ask(rt, l, r);
	return res;
}
int main() {
#ifdef ONLINE_JUDGE
	std::ios::sync_with_stdio(false);
	std::cin.tie(nullptr), std::cout.tie(nullptr);
#else
	std::freopen(".in", "r", stdin);
	std::freopen(".out", "w", stdout);
#endif
	int n, m, q;
	std::cin >> n >> m >> q;
	std::vector<int> fa(n + 1);
	std::vector<std::vector<int> > g(n + 1);
	for (int i = 2; i <= n; ++i) {
		std::cin >> fa[i];
		g[fa[i]].push_back(i);
	}
	for (int i = 1; i <= n; ++i)
		std::cin >> a[i];
	std::vector<int> son(n + 1), siz(n + 1), dep(n + 1);
	std::function<void(int)> DFS = [&](int x) {
		siz[x] = 1;
		for (auto i : g[x]) {
			fa[i] = x;
			dep[i] = dep[x] + 1;
			DFS(i);
			if (siz[i] > siz[son[x]])
				son[x] = i;
			siz[x] += siz[i];
		}
		return;
	};
	dep[1] = 1, DFS(1);
	std::vector<int> top(n + 1), dfn(n + 1);
	std::function<void(int, int)> DFS1 = [&](int x, int fa) {
		static int now = 0;
		top[x] = fa;
		dfn[x] = ++now, tab[now] = x;
		if (son[x])
			DFS1(son[x], fa);
		for (auto i : g[x])
			if (i != son[x])
				DFS1(i, i);
		return;
	};
	DFS1(1, 1), bld(1, 1, n);
	auto getLCA = [&](int x, int y) {
		for (; top[x] != top[y]; x = fa[top[x]])
			if (dep[top[x]] < dep[top[y]])
				std::swap(x, y);
		return dep[x] < dep[y] ? x : y;
	};
	auto ask = [&](int x, int y) {
		std::bitset<1001> res;
		for (; top[x] != top[y]; x = fa[top[x]]) {
			if (dep[top[x]] < dep[top[y]])
				std::swap(x, y);
			res |= ::ask(1, dfn[top[x]], dfn[x]);
		}
		if (dep[x] < dep[y])
			std::swap(x, y);
		return res | ::ask(1, dfn[y], dfn[x]);
	};
	for (int k; q--; ) {
		std::cin >> k;
		std::vector<int> p(k);
		std::vector<std::bitset<1001> > u(k);
		int fa = -1, siz = 1 << k;
		for (int i = 0; i < k; ++i) {
			std::cin >> p[i];
			fa = (~fa ? getLCA(fa, p[i]) : p[i]);
		}
		for (int i = 0; i < k; ++i)
			u[i] = ask(p[i], fa);
		int res = inf;
		for (int i = 1; i < siz; ++i) {
			int cnt = 0;
			std::bitset<1001> now;
			for (int j = 0; j < k; ++j)
				if ((i >> j) & 1)
					now |= u[j], ++cnt;
			res = std::min(res, (int)now.count() / cnt);
		}
		std::cout << res * k << '\n';
	}
	return 0;
}

C. 胖

http://222.180.160.110:61234/contest/5636/problem/3

把边的松弛转化为点的松弛。直觉理解一下，我们发现一个很明显的事实：对于每个修了路的点，其可以松弛到的点是连续的一段区间，一个点被区间覆盖了多少次就代表被松弛了多少次。

不妨考虑对于一个修了路的点 \(i\)，其覆盖到的区间内点 \(x\) 满足的要求。易发现：

  1. 对于任意 \(i<j<x\) 或 \(x<j<i\)，应有 \(d_i+dis(i,j)<d_j\)；
  2. 对于任意 \(i<x<j\) 或 \(j<x<i\)，应有 \(d_i+dis(i,x)<d_j+dis(x,j)\)。

以 \(x>i\) 为例，由 1 可得 \(d_i+s_j-s_i<d_j\)，移项得 \(d_i-s_i<d_j-s_j\)，把 \(d_i-s_i\) 塞到线段树里，就可以二分出仅在条件 1 约束下的右端点。同理可得到左端点。

考虑条件 2。即 \(d_i+s_x-s_i<d_j+s_j-s_x\)，移项得 \(d_i-s_i<d_j+s_j-2\times s_x\)。显然作为可行性右端点，不管怎么说是需要有单调性的。把 \(d_i-s_i\) 和 \(s_i\) 全部塞到线段树里面，把条件 1 得到的左右端点作为二分边界一通二分即可。
