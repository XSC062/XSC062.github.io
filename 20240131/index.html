namespace XSC062 {
using namespace fastIO;
const int maxn = 1e5 + 5;
int n, x, y;
int cnt[maxn];
std::vector<int> g[maxn];
int a[maxn], f[maxn], p[maxn];
int min(int x, int y) {
	return x < y ? x : y;
}
void DFS(int x, int fa) {
	if ((int)g[x].size() == 1)
		cnt[x] = 1;
	for (auto i : g[x]) {
		if (i == fa) continue;
		DFS(i, x);
		f[x] += f[i];
		cnt[x] += cnt[i];
	}
	if (f[x] < cnt[x])
		f[x] = min(f[x] + a[x], cnt[x]);
	return;
}
void add(int x, int y) {
	g[x].push_back(y);
	return;
}
int main() {
	read(n);
	for (int i = 1; i <= n; ++i) read(p[i]);
	for (int i = 1; i < n; ++i) {
		read(x), read(y);
		add(x, y), add(y, x);
	}
	int tot = 0;
	for (int i = 2; i <= n; ++i)
		tot += ((int)g[i].size() == 1);
	for (int i = 1; i <= tot; ++i) ++a[p[i]];
	DFS(1, -1);
	print(f[1], '\n');
	return 0;
}
} // namespace XSC062

const int maxn = 1e4 + 5;

int n, res, mn, tot;
int a[maxn], b[maxm];
bool check(int x) {
	int l1 = 0, l2 = 0, l3 = 0;
	for (int i = 1; i <= n; ++i) {
		if (l1 == 0) l1 = a[i];
		else if (x && (a[i] - l1) % x == 0) l1 = a[i];
		else if (l2 == 0) l2 = a[i];
		else if (x && (a[i] - l2) % x == 0) l2 = a[i];
		else if (l3 == 0) l3 = a[i];
		else if (x && (a[i] - l3) % x == 0) l3 = a[i];
		else return 0;
	}
	return 1;
