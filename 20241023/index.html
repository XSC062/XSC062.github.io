mex

http://222.180.160.110:61234/contest/5662/problem/1

全场唯一真实意义上的小清新。

考虑一个合法 mex 序列开头能放的元素，不难发现只有 \(0\) 和 \(1\)。分开讨论：

  1. 对于开头为 \(0\) 的情况，下一个可以是 \(0\) 或 \(1\) 或 \(2\)。

    1. 若下一个是 \(0\)，同上。

    2. 若下一个是 \(1\)，则第三个可以是 \(1\) 或 \(2\) 或 \(3\)。

    3. 若下一个是 \(2\)，则第三个可以是 \(0\) 或 \(2\)。

      1. 若第三个是 \(0\)，则第四个可以是 \(0\) 或 \(2\)。
      2. 若第三个是 \(2\)，则第四个可以是 \(0\) 或 \(2\)。

  2. 对于开头为 \(1\) 的情况，下一个只能是 \(1\)。

其实规律已经很明显了，令第 \(0\) 个元素为 \(-1\)，则该序列应为一段单调不下降且值连续的序列和一段由 \(a+2\) 和 \(a\) 组成的序列拼接而成。

随便 DP 一下即可。

#include <bits/stdc++.h>
const int mod = 1e9 + 7;
struct mint {
	long long x;
	mint(): x() {}
	mint(long long v): x(v) {}
	mint operator* (const mint &q) const {
		return x * q.x % mod;
	};
	mint& operator*= (const mint &q) {
		return *this = *this * q;
	}
	mint operator+ (const mint &q) const {
		return (x + q.x) % mod;
	}
	mint& operator+= (const mint &q) {
		return *this = *this + q;
	}
	mint operator- (const mint &q) const {
		return (x + mod - q.x) % mod;
	}
	long long operator() (void) {
		return x;
	}
};
int main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(nullptr), std::cout.tie(nullptr);
	std::freopen("mex.in", "r", stdin);
	std::freopen("mex.out", "w", stdout);
	int n;
	std::cin >> n;
	std::vector<int> a(n + 1);
	std::vector<mint> base(n + 1);
	for (int i = 1; i <= n; ++i)
		std::cin >> a[i];
	std::vector<mint> f(n + 1), g(n + 1), h(n + 1);
	mint res;
	for (int i = 1; i <= n; ++i) {
		g[a[i]] *= 2;
		if (a[i] >= 2)
			g[a[i]] += f[a[i] - 2] + h[a[i] - 2];
		else if (a[i] == 1)
			g[a[i]] += 1;
		h[a[i]] *= 2;
		if (a[i] + 2 <= n)
			h[a[i]] += g[a[i] + 2];
		f[a[i]] *= 2;
		if (a[i])
			f[a[i]] += f[a[i] - 1];
		else
			f[a[i]] += 1;
	}
	for (int i = 0; i <= n; ++i) {
		res += f[i] + g[i] + h[i];
		// printf("%d: f = %lld, g = %lld, h = %lld\n", i, f[i], g[i], h[i]);
	}
	std::cout << res() << '\n';
	return 0;
}
