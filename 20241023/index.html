const int mod = 1e9 + 7;
struct mint {
	long long x;
	mint(): x() {}
	mint(long long v): x(v) {}
	mint operator* (const mint &q) const {
		return x * q.x % mod;
	};
	mint& operator*= (const mint &q) {
		return *this = *this * q;
	}
	mint operator+ (const mint &q) const {
		return (x + q.x) % mod;
	}
	mint& operator+= (const mint &q) {
		return *this = *this + q;
	}
	mint operator- (const mint &q) const {
		return (x + mod - q.x) % mod;
	}
	long long operator() (void) {
		return x;
	}
};
