#include <bits/stdc++.h>
namespace XSC062 {
const int maxn = 1e3 + 5;
using db = double;
db p, q;
int T, n;
db f[maxn], g[maxn];
int min(int x, int y) {
	return x < y ? x : y;
}
int main() {
	scanf("%d", &T);
	while (T--) {
		scanf("%d %lf %lf", &n, &p, &q);
		f[0] = .0, g[0] = 1.0;
		for (int i = 1; i <= n && i <= 1000; ++i) {
			if (g[i - 1] > f[i - 1]) {
				f[i] = ((1 - p) * q * f[i - 1] + p * g[i - 1]) / (1 - (1 - p) * (1 - q));
				g[i] = ((1 - q) * p * g[i - 1] + q * f[i - 1]) / (1 - (1 - p) * (1 - q));
			}
			else {
				f[i] = ((1 - q) * p * f[i - 1] + (1 - p) * g[i - 1]) / (1 - p * q);
				g[i] = ((1 - p) * q * g[i - 1] + (1 - q) * f[i - 1]) / (1 - p * q);
			}
		}
		printf("%.6lf\n", f[min(n, 1000)]);
	}
	return 0;
}
} // namespace XSC062
int main() {
	XSC062::main();
	return 0;
}
