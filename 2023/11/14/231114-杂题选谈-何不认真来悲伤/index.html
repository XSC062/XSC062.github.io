A. 印章

https://www.luogu.com.cn/problem/P3585

http://222.180.160.110:1024/contest/4435/problem/1

此题乍一看难以下手，但细细思考便可发现，这待覆盖矩阵之最左上角的 X（其中，「上」又是优先于「左」的），必是需要印章之最左上角（这里的「上」，也是优先于「左」的）的 X 来覆盖的。

既然有这一重要信息，我们便可从此处着手，将印章左上角 X 与待覆盖矩阵的左上角对齐进行覆盖，与此同时，判断印章是否超出边界，或是覆盖了空白点。

如此这般，每次找到矩阵中最左上角的未覆盖 X 并尝试覆盖，便可完成这题目。

这样一来，矩阵中的每个 X 只会被印章遍历到一次，是 $\mathcal O(nm)$ 的复杂度。

namespace XSC062 {
using namespace fastIO;
const int maxn = 1e3 + 5;
struct _ {
    int x, y;
    _() {}
    _(int x1, int y1) {
        x = x1, y = y1;
    }
};
char t;
std::vector<_> p;
int T, n, m, c, r, sx, sy;
bool a[maxn][maxn], b[maxn][maxn];
bool check(int x, int y) {
    for (auto i : p) {
        int nx = x + i.x - sx,
            ny = y + i.y - sy;
        if (nx < 1 || ny < 1 ||
                    nx > n || ny > m)
            return 0;
        if (b[nx][ny] || !a[nx][ny])
            return 0;
        b[nx][ny] = 1;
    }
    return b[x][y];
}
int main() {
    scanf("%d", &T);
    while (T--) {
        scanf("%d %d %d %d", &n, &m, &c, &r);
        p.clear(), sx = sy = 0;
        for (int i = 1; i <= n; ++i) {
            for (int j = 1; j <= m; ++j) {
                scanf("%1s", &t);
                a[i][j] = (t == 'x');
                b[i][j] = 0;
            }
        }
        for (int i = 1; i <= c; ++i) {
            for (int j = 1; j <= r; ++j) {
                scanf("%1s", &t);
                if (t == 'x') {
                    if (!sx) sx = i, sy = j;
                    p.emplace_back(i, j);
                }
            }
        }
        for (int i = 1; i <= n; ++i) {
            for (int j = 1; j <= m; ++j) {
                if (a[i][j] && !b[i][j]) {
                    if (!check(i, j)) {
                        puts("NIE");
                        goto noSol;
                    }
                }
            }
        }
        puts("TAK");
        noSol: ;
    }
    return 0;
}
} // namespace XSC062
