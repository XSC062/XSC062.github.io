A. 分糖果 candy

http://222.180.160.110:1024/contest/4143/problem/1

鲜花 🌼

你看这个 A 题。我为什么打了个下饭的标签呢？因为我读错了题还能过掉带多测的样例。我以为是每袋糖数量是 $i\times i$，最后要保证 $\operatorname{mod} \, m$ 的相同数量最多……

总之已加入 下饭操作合集。

很明显的一个做法，我们统计每种数量的糖的袋数，然后假设最后吃成 $i$ 这种数量的最多。

我们枚举 $i$，那么所有 $j>i$ 都可以被吃成 $i$，设数量为 $i$ 的糖有 $s_i$ 袋，那么对于吃成 $i$ 的答案就是 $s_i + \min(n - \sum_{j=1}^i s_j, t)$，拿个前缀和统计一下就好了。$m$ 的范围很大，所以这里我们用 map 来统计 $s$，然后用迭代器来遍历每个元素。因为 $n$ 很小，可能的元素最多又只有 $n$ 个，所以时间上是没问题的。

但是这样做你就会感受到数据人满满的恶意：从第二个点开始全部开满 $3\times 10^6$，map 直接被卡飞。

所以我们可以转换一下思路，直接排序，然后统计连续相同元素数量。最后时间复杂度 $\mathcal O(c \cdot n\log n)$。

#define int long long
namespace XSC062 {
using namespace fastIO;
const int maxn = 3e5 + 5;
int a[maxn];
int T, n, m, t, ans;
int max(int x, int y) { return x > y ? x : y; }
int min(int x, int y) { return x < y ? x : y; }
int main() {
    read(T);
    while (T--) {
        read(n), read(m), read(t);
        for (int i = 0; i < n; ++i)
            a[i + 1] = (i * i) % m + 1;
        ans = 0;
        std::sort(a + 1, a + n + 1);
        int sum = 0, cnt = 0;
        for (int i = 1; i <= n + 1; ++i) {
            if (a[i] != a[i - 1]) {
                sum += cnt;
                int res = cnt + min(t, n - sum);
                ans = max(ans, res);
                cnt = 0;
            }
            ++cnt;
        }
        print(ans, '\n');
    }
    return 0;
}
} // namespace XSC062
#undef int

B. 破译密码 password

http://222.180.160.110:1024/contest/4143/problem/2

一眼题，对于每个字符串，把里面所有字母并查集合并一下就好了。

注意到并查集关系的传递性，只用对相邻两个合并。

最后在出现过的字母中统计为并查集的根的字母数量即为答案。

namespace XSC062 {
constexpr int maxm = 35;
constexpr int maxn = 1e6 + 5;
int f[maxm];
char a[maxn];
bool is[maxm];
int n, l, cnt;
int fun(int i) {
    return a[i] - 'a' + 1;
}
int find(int x) {
    return x == f[x] ? x : f[x] = find(f[x]);
}
void merge(int x, int y) {
    f[find(x)] = find(y);
    return;
}
int main() {
    scanf("%d", &n);
    for (int i = 1; i <= 26; ++i)
        f[i] = i;
    for (int i = 1; i <= n; ++i) {
        scanf("%s", a + 1);
        l = strlen(a + 1);
        for (int j = 1; j <= l; ++j) {
            is[fun(j)] = 1;
            if (j > 1)
                merge(fun(j), fun(j - 1));
        }
    }
    for (int i = 1; i <= 26; ++i) {
        if (is[i] && f[i] == i)
            ++cnt;
    }
    printf("%d\n", cnt);
    return 0;
}
} // namespace XSC062
