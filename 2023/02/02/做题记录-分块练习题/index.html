A. 教主的魔法

http://222.180.160.110:1024/contest/3213/problem/1

这是什么？数列分块入门 2，秒一下。

这里给出一个只用改动很少的方案：将所有输入取反，再将 std::lower_bound 替换为 std::upper_bound 即可。

namespace XSC062 {
using namespace fastIO;
const int maxk = 1e5 + 5;
const int maxn = 1e6 + 5;
struct _ {
    int l, r, d;
};
_ t[maxk];
char op;
int n, m, k, siz, l, r, x;
int a[maxn], f[maxn], b[maxn];
inline int min(int x, int y) {
    return x < y ? x : y;
}
inline int max(int x, int y) {
    return x > y ? x : y;
}
inline void order(int l, int r) {
    for (int i = l; i <= r; ++i)
        b[i] = a[i];
    std::sort(b + l, b + r + 1);
    return;
}
inline void Init(int n) {
    siz = sqrt(n / log(n) * log(2.0));  
    k = (n + siz - 1) / siz;
    for (int i = 1; i <= k; ++i) {
        t[i].l = t[i - 1].r + 1;
        t[i].r = t[i].l + siz - 1;
    }
    t[k].r = n;
    for (int i = 1; i <= n; ++i)
        f[i] = (i + siz - 1) / siz;
    for (int i = 1; i <= k; ++i)
        order(t[i].l, t[i].r);
    return;
}
inline void add(int l, int r, int x) {
    for (int i = f[l] + 1; i < f[r]; ++i)
        t[i].d += x;
    int lim = min(t[f[l]].r, r);
    for (int i = l; i <= lim; ++i)
        a[i] += x;
    order(t[f[l]].l, t[f[l]].r);
    if (f[l] != f[r]) {
        for (int i = t[f[r]].l; i <= r; ++i)
            a[i] += x;
        order(t[f[r]].l, t[f[r]].r);
    }
    return;
}
inline int qry(int l, int r, int x) {
    int res = 0, u;
    for (int i = f[l] + 1; i < f[r]; ++i) {
        u = std::upper_bound(b + t[i].l,
            b + t[i].r + 1, x - t[i].d) - b - 1;
        if (u >= t[i].l)
            res += u - t[i].l + 1;
    }
    int lim = min(t[f[l]].r, r);
    for (int i = l; i <= lim; ++i)
        res +=  (a[i] + t[f[l]].d <= x);
    if (f[l] != f[r]) {
        for (int i = t[f[r]].l; i <= r; ++i)
            res += (a[i] + t[f[r]].d <= x);
    }
    return res;
}
int main() {
    read(n), read(m);
    for (int i = 1; i <= n; ++i)
        read(a[i]), a[i] = -a[i];
    Init(n);
    while (m--) {
        scanf("%1s", &op);
        read(l), read(r), read(x);
        if (op == 'M')
            add(l, r, -x);
        else print(qry(l, r, -x), '\n');
    }
    return 0;
}
} // namespace XSC062

懂了，去屑，我用海某丝。

一个静态问题。如果不强制在线那莫队乱杀，可惜换不得。

参考数列分块入门 9，这题大概率是需要预处理了。那我们预处理的东西也可以参考一下，是否能够处理整块 $i$
到整块 $j$ 中的答案？

盒盒盒那么这个预处理就可以莫队了，你看，就算强制在线莫队也能乱杀。

我们已经知道了整块的答案，那么不妨考虑，什么会对答案产生影响？

答案是在散点中出现的数。

于是我们还是记录每个数出现过的位置，然后若其在整块中出现偶数次，但在整个询问中出现奇数次，则答案减去之；若其在整块中出现奇数次，但在整个询问中出现了偶数次，则答案加上之。
