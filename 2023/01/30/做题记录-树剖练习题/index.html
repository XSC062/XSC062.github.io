A. LCA

http://222.180.160.110:1024/contest/3211/problem/1

为方便描述，编号统一加 1。

注意这里的 $l\sim r$ 彼此之间是没有关联的。

暴力 LCA 肯定是不行的。不难想到每一个询问可以树剖，清空线段树再暴力枚举 $l\sim r$，将根节点与节点的路径区间加 1，再区间询问根节点到 $z$ 的路径。时间复杂度 $\mathcal O(q\times n\times \log^2n)$。

然后是思路转换点。我们会单个询问，但单次的复杂度就很逆天了，再加之这是一个区间修改的 DS 题，联想到离线。

怎么离线呢？这就很好想了，运用差分思想，先按右端点 $r$ 排序，将 $1\sim r$ 的点到根节点的路径全部加 1，询问根节点到 $z$，得到第一步答案。

接下来清空线段树，按左端点 $l$ 排序，将 $1\sim l-1$ 到根节点的路径全部减 1，询问根节点到 $z$，加之以第一步的答案，得到最终结果。

namespace XSC062 {
using namespace fastIO;
const int inf = 1e18;
const int mod = 201314;
const int maxn = 5e4 + 5;
#define lt (p << 1)
#define rt (lt | 1)
struct _ {
    int l, r;
    int u, d;
};
struct __ {
    int res;
    int l, r, z, i;
};
__ r[maxn];
_ t[maxn << 2];
int n, x, q, now, p;
std::vector<int> g[maxn];
int a[maxn], siz[maxn], fa[maxn];
int tab[maxn], ten[maxn], dep[maxn];
int wei[maxn], top[maxn], dfn[maxn];
inline void pushup(int p) {
    t[p].u = (t[lt].u + t[rt].u) % mod;
    return;
}
inline void swap(int &x, int &y) {
    x ^= y ^= x ^= y;
    return;
}
inline void pushdown(int p) {
    if (t[p].d) {
        (t[lt].d += t[p].d) %= mod;
        (t[rt].d += t[p].d) %= mod;
        (t[lt].u += t[p].d
            * (t[lt].r - t[lt].l + 1)) %= mod;
        (t[rt].u += t[p].d
            * (t[rt].r - t[rt].l + 1)) %= mod;
        t[p].d = 0;
    }
    return;
}
void bld(int p, int l, int r) {
    t[p].d = t[p].u = 0;
    t[p].l = l, t[p].r = r;
    if (l == r)
        return;
    int mid = (l + r) >> 1;
    bld(lt, l, mid);
    bld(rt, mid + 1, r);
    return;
}
void upd(int p, int l, int r, int u) {
    if (l <= t[p].l && t[p].r <= r) {
        (t[p].d += u) %= mod;
        (t[p].u += (t[p].r - t[p].l + 1) * u) %= mod;
        return;
    }
    int mid = (t[p].l + t[p].r) >> 1;
    pushdown(p);
    if (l <= mid)
        upd(lt, l, r, u);
    if (r > mid)
        upd(rt, l, r, u);
    pushup(p);
    return;
}
int qsum(int p, int l, int r) {
    if (l <= t[p].l && t[p].r <= r)
        return t[p].u;
    int res = 0;
    int mid = (t[p].l + t[p].r) >> 1;
    pushdown(p);
    if (l <= mid)
        res = qsum(lt, l, r) % mod;
    if (r > mid)
        (res += qsum(rt, l, r)) %= mod;
    return res;
}
void DFS1(int x, int f) {
    wei[x] = 0;
    fa[x] = f, siz[x] = 1;
    for (auto i : g[x]) {
        if (i == f)
            continue;
        dep[i] = dep[x] + 1;
        DFS1(i, x);
        siz[x] += siz[i];
        if (siz[i] > siz[wei[x]])
            wei[x] = i;
    }
    return;
}
void DFS2(int x, int t) {
    top[x] = t;
    dfn[++now] = x;
    tab[x] = now;
    if (wei[x])
        DFS2(wei[x], t);
    for (auto i : g[x]) {
        if (i == fa[x] || i == wei[x])
            continue;
        DFS2(i, i);
    }
    ten[x] = now;
    return;
}
inline void add(int x, int y, int u) {
    while (top[x] != top[y]) {
        if (dep[top[x]] < dep[top[y]])
            swap(x, y);
        upd(1, tab[top[x]], tab[x], u);
        x = fa[top[x]];
    }
    if (dep[x] < dep[y])
        swap(x, y);
    upd(1, tab[y], tab[x], u);
    return;
}
inline int qry(int x, int y) {
    int res = 0;
    while (top[x] != top[y]) {
        if (dep[top[x]] < dep[top[y]])
            swap(x, y);
        (res += qsum(1, tab[top[x]], tab[x])) %= mod;
        x = fa[top[x]];
    }
    if (dep[x] < dep[y])
        swap(x, y);
    (res += qsum(1, tab[y], tab[x])) %= mod;
    return res;
}
inline void add(int x, int y) {
    g[x].push_back(y);
    return;
}
int main() {
    read(n), read(q);
    for (int i = 2; i <= n; ++i) {
        read(x), ++x;
        add(i, x), add(x, i);
    }
    dep[1] = 1;
    DFS1(1, -1), DFS2(1, 1);
    for (int i = 1; i <= q; ++i) {
        read(r[i].l), read(r[i].r);
        read(r[i].z), r[i].i = i;
        ++r[i].l, ++r[i].r, ++r[i].z;
    }
    bld(1, 1, n);
    std::sort(r + 1, r + q + 1, [&](__ x, __ y)
                -> bool { return x.r < y.r; } );
    p = 0;
    for (int i = 1; i <= q; ++i) {
        while (p < r[i].r)
            add(1, ++p, 1);
        r[i].res = qry(1, r[i].z);
    }
    bld(1, 1, n);
    std::sort(r + 1, r + q + 1, [&](__ x, __ y)
                -> bool { return x.l < y.l; } );
    p = 0;
    for (int i = 1; i <= q; ++i) {
        while (p < r[i].l - 1)
            add(1, ++p, -1);
        (r[i].res += qry(1, r[i].z) + mod) %= mod;
    }
    std::sort(r + 1, r + q + 1, [&](__ x, __ y)
                -> bool { return x.i < y.i; } );
    for (int i = 1; i <= q; ++i)
        print(r[i].res, '\n');
    return 0;
}
} // namespace XSC062
